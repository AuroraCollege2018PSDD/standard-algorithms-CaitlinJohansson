	CASEWHERE choice is
	  L: LoadArray

	  Q: continue = False
	ENDCASE
  END WHILE

BEGIN LoadArray
  Let i = 1
  Read DataValue
  WHILE DataValue <> "xxx"
    Let Element (i) = DataValue
	i = i + 1
	Read DataValue
  ENDWHILE
  Let NumElements = i
  Display "There are" NumElements "items loaded into the array"
END LoadArray

BEGIN PrintArrayContents
  let i = 1
  REPEAT
    Display Element(i)
	i = i + 1
  UNTIL i >= NumElements
END printArray

BEGIN SumArrayContents
  Let i = 1
  Let total = 0
  REPEAT
    Let total = Element (i) + total
    i = i + 1
  UNTIL i > NumElements
  Display "The sum of all the elements in the array = " total
END SumArrayContents
